git config --global user.name <userName>  <----- set user name
 
git config --global user.email <userEmail> <----- set email id

git init <----- to set repository first time

git add <name of file><----- to track files in staged area
or
git add -A <------ to add file for tracking in single go

git commit -m <"message"> <----------- initial commit need to do fisrt time

git status <---- shows status 

git diff <----- compare and show differnce between modified and staged file

git commit -a -m <"message"> <-- you can skip staged area and directly commit(NOT GOOD)


touch <filename.extenation><---- to add empty file in repo

git rm --cached <filename.extension> <--- remove file from tracking/staged area but keep in HDD

git rm <filename.extension> <---- delete file from stage/tracking area and from HDD too
